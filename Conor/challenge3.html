        import java.util.Scanner;
        public class boatSeats {
            public static void main(String[] args) {
                Scanner scnr = new Scanner(System.in);
                int total = 0;
                int availableSeats;
                int seatRows;
        
                seatRows = scnr.nextInt();
        
                for (int i = 0; i < seatRows; ++i) {
                    availableSeats = scnr.nextInt();
                    total = total + availableSeats;
                }
        
                System.out.println("There are " + total + " seats left on the escape boat.");
        
            }  
        }
